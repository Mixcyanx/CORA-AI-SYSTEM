// ========================================
// REQUEST KIND DETECTION
// ========================================
function detectRequestKind(question) {

  const text =
    String(question || "")
      .trim()
      .toLowerCase();

  if (!text) {
    return "general";
  }

  if (/(幾點|現在時間|目前時間|today'?s time|current time|what time|時間)/i.test(text)) {
    return "time";
  }

  if (/(生成|產生|畫|做一張|設計|create|generate|draw|image|picture|logo|封面|去背|圖片)/i.test(text) &&
      !/(分析|辨識|這張|這個藥|藥品|藥物資訊|成分)/i.test(text)) {
    return "image_generation";
  }

  if (/(藥|藥品|藥物|用藥|副作用|成分|適應症|禁忌|劑量|膠囊|錠|錠劑|藥師|處方|medicine|drug|pill|tablet|capsule|side effect)/i.test(text)) {
    return "drug";
  }

  return "general";
}

// ========================================
// TAIPEI TIME TOOL
// ========================================
function getTaipeiTimeAnswer(question) {

  const now = new Date();

  const parts =
    new Intl.DateTimeFormat("zh-TW", {
      timeZone: "Asia/Taipei",
      year: "numeric",
      month: "2-digit",
      day: "2-digit",
      weekday: "long",
      hour: "2-digit",
      minute: "2-digit",
      second: "2-digit",
      hour12: false
    }).format(now);

  return `台灣時間現在是 ${parts}。`;
}

// ========================================
// IMAGE GENERATION
// ========================================
async function generateImagesWithChatGPT(
  prompt,
  chatgptApi,
  chatgptToken
) {

  try {

    const imageApi =
      String(chatgptApi)
        .replace(/\/chat\/completions\/?$/, "/images/generations");

    const res =
      await fetch(imageApi, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "Authorization": chatgptToken
        },
        body: JSON.stringify({
          model: "gpt-image-1",
          prompt: String(prompt || "").slice(0, 1800),
          size: "1024x1024",
          n: 1
        })
      });

    const data =
      await safeJson(res);

    const images =
      (data.data || [])
        .map(item => {
          if (item.url) return item.url;
          if (item.b64_json) return `data:image/png;base64,${item.b64_json}`;
          return "";
        })
        .filter(Boolean);

    if (images.length) {
      return { images };
    }

    return {
      images: [],
      message: data.error?.message || data.raw || "圖片 API 未回傳圖片。"
    };

  } catch (err) {

    return {
      images: [],
      message: `圖片生成暫時失敗：${err.message}`
    };
  }
}

// ========================================
// CROSS-CHAT PERSONAL MEMORY
// ========================================
async function getPersonalMemoryContext(
  projectId,
  idToken,
  uid,
  currentConversationId
) {

  try {

    const conversations =
      await listConversations(
        projectId,
        idToken,
        uid
      );

    const recent =
      conversations
        .filter(chat => normalizeConversationId(chat.id) !== normalizeConversationId(currentConversationId))
        .slice(0, 12);

    const blocks = [];

    for (const chat of recent) {

      try {

        const full =
          await loadConversation(
            projectId,
            idToken,
            uid,
            chat.id
          );

        const memory =
          Array.isArray(full.chatMemory) && full.chatMemory.length
            ? full.chatMemory
            : (Array.isArray(full.messages) ? full.messages : [])
                .filter(m => m.role === "user" || m.role === "assistant")
                .slice(-6)
                .map(m => ({
                  role: m.role,
                  content: m.content || ""
                }));

        if (memory.length) {
          blocks.push(
            `【${full.title || chat.title || "未命名對話"}】\n` +
            memory
              .slice(-6)
              .map(m => `${m.role === "user" ? "使用者" : "CORA"}：${String(m.content || "").slice(0, 240)}`)
              .join("\n")
          );
        }

      } catch {
        // Ignore unreadable old conversations.
      }
    }

    return blocks
      .join("\n---\n")
      .slice(0, 6000);

  } catch {

    return "";
  }
}

// ========================================
// CLEAN TITLE
// ========================================
function cleanTitle(title) {

  return String(title || "新的對話")
    .replace(/[\\n\\r"「」]/g, "")
    .trim()
    .slice(0, 18) || "新的對話";
}

// ========================================
// DEDUPE
// ========================================
function dedupeMessages(messages) {

  const seen =
    new Set();

  const result = [];

  for (const msg of messages || []) {

    const key =
      `${msg.role}|${msg.content || ""}|${JSON.stringify(msg.images || [])}`;

    if (seen.has(key)) {
      continue;
    }

    seen.add(key);

    result.push(msg);
  }

  return result;
}

// VERIFY FIREBASE
// ========================================
async function verifyFirebaseUser(
  idToken,
  apiKey
) {

  if (!idToken) {
    throw new Error("尚未登入");
  }

  const res =
    await fetch(
      `https://identitytoolkit.googleapis.com/v1/accounts:lookup?key=${apiKey}`,
      {
        method: "POST",
        headers: {
          "Content-Type":
            "application/json"
        },
        body: JSON.stringify({
          idToken
        })
      }
    );

  const data =
    await safeJson(res);

  if (!data.users?.[0]) {

    throw new Error(
      "Firebase驗證失敗"
    );
  }

  return data.users[0].localId;
}

// ========================================
// FIRESTORE BASE
// ========================================
function firestoreBase(projectId) {

  return `https://firestore.googleapis.com/v1/projects/${projectId}/databases/(default)/documents`;
}

// ========================================
// NORMALIZE ID
// ========================================
function normalizeConversationId(id) {

  const raw =
    String(id || "").trim();

  if (
    !raw ||
    raw.startsWith("local-")
  ) {

    return "";
  }

  return raw
    .split("/")
    .pop();
}

// ========================================
// SAVE CONVERSATION
// ========================================
async function saveConversation(
  projectId,
  idToken,
  uid,
  conversation
) {

  const conversationId =
    normalizeConversationId(
      conversation.id
    ) || crypto.randomUUID();

  const payload = {
    fields: {

      title: {
        stringValue:
          conversation.title || "新對話"
      },

      messages: {
        stringValue:
          JSON.stringify(
            dedupeMessages(
              conversation.messages || []
            )
          )
      },

      chatMemory: {
        stringValue:
          JSON.stringify(
            conversation.chatMemory || []
          )
      },

      medicationInfo: {
        stringValue:
          JSON.stringify(
            conversation.medicationInfo || {}
          )
      },

      geminiChatId: {
        stringValue:
          conversation.geminiChatId || ""
      },

      messageCount: {
        integerValue:
          String(
            (conversation.messages || []).length
          )
      },

      updatedAt: {
        timestampValue:
          new Date().toISOString()
      },

      lastAccessedAt: {
        timestampValue:
          new Date().toISOString()
      }
    }
  };

  const url =
    `${firestoreBase(projectId)}` +
    `/users/${uid}/conversations/${conversationId}`;

  const res =
    await fetch(url, {
      method: "PATCH",
      headers: {
        "Content-Type":
          "application/json",

        "Authorization":
          `Bearer ${idToken}`
      },
      body: JSON.stringify(payload)
    });

  const data =
    await safeJson(res);

  if (!res.ok) {

    throw new Error(
      data.error?.message ||
      "儲存對話失敗"
    );
  }

  return {
    conversationId
  };
}

// ========================================
// LIST CONVERSATIONS
// ========================================
async function listConversations(
  projectId,
  idToken,
  uid
) {

  const url =
    `${firestoreBase(projectId)}` +
    `/users/${uid}/conversations?pageSize=100`;

  const res =
    await fetch(url, {
      headers: {
        "Authorization":
          `Bearer ${idToken}`
      }
    });

  const data =
    await safeJson(res);

  const rows = (
    data.documents || []
  ).map(doc => {

    const fields =
      doc.fields || {};

    return {
      id:
        doc.name.split("/").pop(),

      title:
        fields.title?.stringValue || "未命名",

      messageCount:
        Number(
          fields.messageCount?.integerValue || 0
        ),

      geminiChatId:
        fields.geminiChatId?.stringValue || "",

      updatedAt:
        fields.updatedAt?.timestampValue || "",

      lastAccessedAt:
        fields.lastAccessedAt?.timestampValue || "",

      createdAt:
        fields.createdAt?.timestampValue || ""
    };
  });

  return dedupeConversations(rows)
    .sort((a, b) =>
      getConversationTime(b) -
      getConversationTime(a)
    );
}

// ========================================
// DEDUPE CONVERSATIONS
// ========================================
function getConversationTime(chat) {

  const raw =
    chat?.lastAccessedAt ||
    chat?.updatedAt ||
    chat?.createdAt ||
    0;

  const parsed =
    typeof raw === "number"
      ? raw
      : Date.parse(raw);

  return Number.isFinite(parsed)
    ? parsed
    : 0;
}

function dedupeConversations(conversations) {

  const map =
    new Map();

  for (const chat of conversations || []) {

    const id =
      normalizeConversationId(chat?.id);

    if (!id) {
      continue;
    }

    const normalized = {
      ...chat,
      id
    };

    const old =
      map.get(id);

    if (
      !old ||
      getConversationTime(normalized) >=
        getConversationTime(old)
    ) {
      map.set(id, normalized);
    }
  }

  return [...map.values()];
}

// CHECK GEMINIDATA USEFUL DATA
// ========================================
function hasUsefulGeminiData(
  graphSummary,
  rawText
) {

  const text =
    String(graphSummary || rawText || "")
      .trim()
      .toLowerCase();

  if (!text) {
    return false;
  }

  const emptySignals = [
    "目前沒有查詢結果",
    "沒有查詢結果",
    "目前沒有找到",
    "沒有找到",
    "查無資料",
    "找不到資料",
    "找不到相關資料",
    "無相關資料",
    "沒有相關資料",
    "暫無資料",
    "資料庫暫無",
    "no result",
    "no results",
    "not found",
    "no data",
    "no relevant",
    "no useful",
    "empty",
    "null",
    "undefined",
    "error",
    "failed",
    "[]",
    "{}"
  ];

  if (
    emptySignals.some(signal =>
      text === signal ||
      text.includes(signal)
    )
  ) {
    return false;
  }

  if (text.length < 12) {
    return false;
  }

  return true;
}

// ========================================
// PARSE JSON
// ========================================
function parseJsonFromLLM(text) {

  try {

    let clean =
      String(text || "")
        .replace(/```json/g, "")
        .replace(/```/g, "")
        .trim();

    const firstBrace =
      clean.indexOf("{");

    const lastBrace =
      clean.lastIndexOf("}");

    if (
      firstBrace >= 0 &&
      lastBrace > firstBrace
    ) {

      clean =
        clean.slice(
          firstBrace,
          lastBrace + 1
        );
    }

    return JSON.parse(clean);

  } catch {

    return {
      answer: text
    };
  }
}

function normalizeDrugCard(card) {

  const source =
    card || {};

  return {

    ingredients:
      toArray(source.ingredients),

    indications:
      toArray(source.indications),

    side_effects:
      toArray(
        source.side_effects ||
        source.sideEffects
      ),

    warnings:
      toArray(source.warnings),

    reference_image_query:
      source.reference_image_query ||
      "",

    image_exploration:
      toArray(
        source.image_exploration
      )
  };
}

// ========================================
// TO ARRAY
// ========================================
function toArray(value) {

  if (!value) {

    return [];
  }

  if (Array.isArray(value)) {

    return value
      .map(v => String(v).trim())
      .filter(Boolean);
  }

  return String(value)
    .split(/\n|、|;|；|,/)
    .map(v => v.trim())
    .filter(Boolean);
}

// ========================================
// SAFE JSON
// ========================================
async function safeJson(response) {

  const text =
    await response.text();

  try {

    return JSON.parse(text);

  } catch {

    return {
      raw: text
    };
  }
}

// ========================================
// GET TOKEN
// ========================================
function getBearerToken(request) {

  const auth =
    request.headers.get(
      "Authorization"
    ) || "";

  if (
    auth.startsWith("Bearer ")
  ) {

    return auth.slice(7);
  }

  return null;
}
